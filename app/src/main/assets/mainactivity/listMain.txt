import android.support.v7.app.AppCompatActivity
import android.os.Bundle
import android.widget.ArrayAdapter
import android.widget.ListView
import android.widget.Toast

class com.example.androidkotlin.MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val listView = findViewById<ListView>(R.id.list) as ListView

        val fruits = arrayOf("Grapes","Pineapple","Apple","Kiwi","Mango","Blueberry","Strawberry","Watermelon","Orange","Banana","Guava","Lime","Avocado","Papaya")

        val adp : ArrayAdapter<String> = ArrayAdapter(this@com.example.androidkotlin.MainActivity,android.R.layout.simple_list_item_1,fruits)

        listView.adapter = adp

        listView.setOnItemClickListener { parent, view, position, id ->

            Toast.makeText(this@com.example.androidkotlin.MainActivity, " Item Clicked:- $id", Toast.LENGTH_SHORT).show()

        }

    }
}
