import android.support.v7.app.AppCompatActivity
import android.os.Bundle
import android.widget.TextView
import android.widget.Toast

class com.example.androidkotlin.MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val helloText = findViewById<TextView>(R.id.textHello) as TextView

        helloText.setOnClickListener {

            Toast.makeText(this@com.example.androidkotlin.MainActivity,"Hello Kotlin",Toast.LENGTH_SHORT).show()

        }
    }
}
